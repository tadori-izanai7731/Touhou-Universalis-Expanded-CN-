country_event = {
	id = KKO_qin.2
	title = KKO_qin.2.t
	desc = KKO_qin.2.d
	picture = kko_mission_12
	
	trigger = {
		NOT = { has_country_flag = kko_xiwang_lose }
		tag = KKO
		OR = {
			AND = {
				has_country_flag = th_qin_started_the_circle
				had_country_flag = {
					flag = q_had_tiangou
					days = 730
				}
			}
			has_country_flag = th_qin_first_circle
		}
		NOT = { has_country_flag = th_qin_have_event }
	}
	
	mean_time_to_happen = {
		months = 1
	}
	
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					has_country_flag = th_qin_first_circle
				}
				clr_country_flag = th_qin_first_circle
			}
			clr_country_flag = q_had_tiangou
			set_country_flag = th_qin_have_event
		}
	}
	
	after = {
		clr_country_flag = th_qin_have_event
	}
	
	option = {
		name = KKO_qin.2.a
		th_qin_set_cycle = { type = huonan }
	}
}
