	picture = ""
	trigger = {
		tag = IQU
		has_country_flag = THE_IQU_now_is_exploring_flag
		OR = {
			NOT = { has_country_flag = THE_IQU_had_explore_event }
			had_country_flag = {
				flag = THE_IQU_had_explore_event
				days = 730
			}
		}
		any_owned_province = {
			area = Dragon_islands
			has_province_flag = THE_IQU_this_provinces_is_exploring_now
		}
	}
	immediate = {
		clr_country_flag = THE_IQU_had_explore_event
		set_country_flag = THE_IQU_having_explore_event
	}
	after = {
		set_country_flag = THE_IQU_had_explore_event
		clr_country_flag = THE_IQU_having_explore_event
	}
	mean_time_to_happen = {
		months = 24
	}
	option = {
		name = THE_IQU_flavor_events.2.1
		THE_IQU_check_none_normal_change_event_random_effects = {
			ne_id = THE_IQU_flavor_events.5
		}
	}
	option = {
		name = THE_IQU_flavor_events.2.2
		THE_IQU_check_nautical_experience_change_event_random_effects = {
			better_id = THE_IQU_flavor_events.3
			bad_id = THE_IQU_flavor_events.4
		}
	}
