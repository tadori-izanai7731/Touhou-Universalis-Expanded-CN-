					NOT = { has_province_modifier = THE_SEJ_p1_mayan_spread_center_modifier }
				}
			}
			random_owned_province = {
				limit = {

					is_city = yes
					religion = THE_Seija_religion
					NOT = { has_province_modifier = THE_SEJ_p1_mayan_spread_center_modifier }
				}
				change_religion = mesoamerican_religion
				add_permanent_province_modifier = {
					name = THE_SEJ_p1_mayan_spread_center_modifier

				}

		}

					num_of_owned_provinces_with = {
						value = 2

				}

					religion = THE_Seija_religion
					NOT = { has_province_modifier = THE_SEJ_p1_mayan_spread_center_modifier }

			}
