		}

			if = {
				limit = {
					any_country = { religion = mesoamerican_religion }
				}
				any_country = {
					religion = mesoamerican_religion
					has_spy_network_from = {
						who = ROOT
						value = 20
					}
				}
			}
			else = {
				num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
					value = 20
					region = central_america_region
				}
			}
		} 

		effect = {
			add_dip_power = 200
			add_country_modifier = {
				name = THE_SEJ_adaptive_training_modifier
				duration = -1
			}
			custom_tooltip = THE_SEJ_create_60_conquistador_add_6_manuever_tt
			hidden_effect = {
				create_conquistador = {
					tradition = 60
					add_manuever = 6
				}
			}
			custom_tooltip = empty_line_tt
			custom_tooltip = THE_SEJ_add_claim_in_spy_network_country_and_neighbor_tt
			hidden_effect = {
				every_country = {
					limit = {
						religion = mesoamerican_religion
						has_spy_network_from = {
							who = ROOT
							value = 20
						}
					}
					every_owned_province = {
						add_permanent_claim = ROOT
					}
					every_neighbor_country = {
						every_owned_province = {
							add_permanent_claim = ROOT

					}
				}
			}
		}
