THE_IQU_this_province_had_been_explored_custom_trigger = {
	$area$ = {
		custom_trigger_tooltip = {
			tooltip = THE_IQU_this_province_had_been_explored_ctt_1
			hidden_trigger = {
				ROOT = {
					OR = {
						any_owned_province = {
							area = $area$
							has_province_flag = THE_IQU_this_province_had_been_explored_flag
						}
						any_subject_country = {
							any_owned_province = {
								area = $area$
								has_province_flag = THE_IQU_this_province_had_been_explored_flag
							}
						}
					}
				}
			}	
		}
	}
}

THE_IQU_check_province_modifier_trigger = {
	IQU = {
		AND = {
			NOT = {
				any_owned_province = {
					has_province_modifier = $modifier$
				}
			}
			NOT = {
				any_subject_country = {
					any_owned_province = {
						has_province_modifier = $modifier$
					}
				}
			}
		}
	}
}
