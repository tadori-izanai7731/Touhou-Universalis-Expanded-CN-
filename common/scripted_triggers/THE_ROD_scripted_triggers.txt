THE_ROD_have_not_unity_german_culture_scripted_trigger = {
	if = {
		limit = {
			NOT = { has_country_flag = THE_ROD_had_accept_german_$culture$_culture_flag }
		}
		any_owned_province = {
			culture = $culture$
		}
	}
	else = {
		custom_trigger_tooltip = {
			tooltip = ""
			always = no
		}
	}
}

THE_ROD_has_test_flag_amount = {
	AND = {
		calc_true_if = {
			amount = $amount_1$
			has_country_flag = THE_ROD_test_1_flag
			has_country_flag = THE_ROD_test_2_flag
			has_country_flag = THE_ROD_test_3_flag
			has_country_flag = THE_ROD_test_4_flag
			has_country_flag = THE_ROD_test_5_flag
			has_country_flag = THE_ROD_test_6_flag
			has_country_flag = THE_ROD_test_7_flag
		}
		NOT = {
			calc_true_if = {
				amount = $amount_2$

				has_country_flag = THE_ROD_test_2_flag

			}
		}
	}
}
