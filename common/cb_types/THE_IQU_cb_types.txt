	valid_for_subject = no

	prerequisites_self = { 
		has_country_flag = THE_IQU_enable_england_war_cb_flag
	}
	prerequisites = {
		FROM = {
			OR = {
				any_owned_province = {
					has_port = yes
				}
				any_subject_country = {
					any_owned_province = {
						has_port = yes
					}
				}
			}
			is_subject = no
		}
		any_province = {
			OR = {
				region = british_isles_region
				region = scandinavia_region
				region = iberia_region
				region = france_region
				region = maghreb_region
				region = low_countries_region
				region = canada_region
				region = north_german_region
			}
			owned_by = FROM
		}
	}

	 war_goal = THE_IQU_england_war_goals
}
