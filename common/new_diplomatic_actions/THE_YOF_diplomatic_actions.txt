the_gather_doom_on = {
	category = th_diplo_action
	require_acceptance = yes
	is_visible = {
		has_government_mechanic = yokari_forest_mechanic
		has_country_flag = gather_doom_on_allow_flag
		FROM = {
			NOT = {
				has_country_flag = THE_YOF_gather_doom_target_flag
			}
		}
	}
	is_allowed = {
		FROM = {
			OR = {
				NOT = { has_country_flag = THE_YOF_cancle_gather_doom_flag }
				had_country_flag = {
					flag = THE_YOF_cancle_gather_doom_flag
					days = 365
				}
			}
		}
	}
	on_accept = {
		hidden_effect = {
			change_variable = {
				which = yof_doom_gather_on_or_off_num
				value = 1
			}
		}
		THE_YOF_refresh_doom_gather_modifier = yes
		FROM = {
			add_country_modifier = {
				name = yof_gather_doom_on_buff2
				duration = 3650
			}
			set_country_flag = THE_YOF_gather_doom_target_flag
		}
	}
	on_decline = {
	}
	ai_acceptance = {
		#<localisation_key>
		#TH_DIPLO_OPPINION
		#基础-30
		add_entry = {
			name = TH_DIPENDENCE
			export_to_variable = {
				variable_name = ai_value
				value = -15
			}
		}
		#同盟+50
		add_entry = {
			name = TH_DIPLO_ALLIENCE
			limit = {
				alliance_with = FROM
			}
			export_to_variable = {
				variable_name = ai_value
				value = 50
			}
		}
		#外交关系+0.2*value
		add_entry = {
			name = TH_DIPLO_OPPINION
			export_to_variable = {
				variable_name = ai_value
				value = opinion
				who = FROM
				with = ROOT
			}
			multiply_variable = {
				which = ai_value 
				value = 0.2
			}
		}
		#宿敌或敌意-50
		add_entry = {
			name = TH_DIPLO_ENERMY
			limit = {
				FROM = {
					OR = {
						ai_attitude = {
							who = ROOT
							attitude = attitude_rivalry
						}
						ai_attitude = {
							who = ROOT
							attitude = attitude_hostile
						}
					}
				}
			}
			export_to_variable = {
				variable_name = ai_value
				value = -50
			}
		}
		#友好+20
		add_entry = {
			name = TH_DIPLO_FRI
			limit = {
				FROM = {
					ai_attitude = {
						who = ROOT
						attitude = attitude_friendly
					}
				}
			}
			export_to_variable = {
				variable_name = ai_value
				value = 20
			}
		}
		#外交声誉+5*value
		add_entry = {
			name = TH_DIPLO_REPULATION
			export_to_variable = {
				variable_name = ai_value
				value = trigger_value:diplomatic_reputation
				who = ROOT
			}
			multiply_variable = {
				which = ai_value 
				value = 5
			}
		}
	}
	ai_will_do = {
		always = no
	}
}

					has_country_flag = the_dirt_river_dip_action_cd
				}
			}
		}
	}
	on_accept = {
		change_variable = {
			which = yof_dirt_river_action_num
			value = 1
		}
		add_spy_network_in = {
			who = from
			value = -15
		}
		add_government_power = {
			mechanic_type = yokari_forest_mechanic
			power_type = yokari_forest_doom_power
			value = -50
		}
		FROM = {
			add_country_modifier = {
				name = yof_dirt_river_action_buff
				duration = 3650
			}
			add_spy_network_in = {
				who = root
				value = -100
			}

		set_country_flag = the_dirt_river_dip_action_cd
	}

	is_visible = {

		}
		FROM = {
			OR = {
				tag = YOF
				was_tag = YOF
			}
		}
	}
	is_allowed = {
		if = {
			limit = { 
				alliance_with = FROM
			}
			treasury = 50
		}
		else_if = {
			limit = { 
				NOT = {
					alliance_with = FROM
				}

					has_opinion = {
						who = root
						value = 100
					}
				}
			}
			treasury = 150
		}
		else = {
			treasury = 250
		}
	}
	on_accept = {
		root = {
			remove_country_modifier = yof_doom_river_flowed_neighbourhood_buff
		}
		if = {
			limit = { 
				alliance_with = FROM
			}
			add_treasury = -50
		}
		else_if = {

					alliance_with = FROM

					has_opinion = {

					}
				}
			}

		}
