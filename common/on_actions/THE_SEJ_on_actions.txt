on_war_ended = {
	THE_SEJ_p0_on_war_ended_scripted_effect = yes
}

on_war_lost = {
	THE_SEJ_magic_charge_war_lost_effect = yes
}

on_province_religion_converted = {
	THE_SEJ_p0_tribunal_conversion_effect = yes
	THE_SEJ_magic_charge_province_religion_converted_effect = yes
	THE_SEJ_p1_mayan_spread_center_conversion_cleanup_effect = yes
}

on_province_owner_change = {
	if = {
		limit = {
			has_province_modifier = THE_SEJ_p1_mayan_spread_center_modifier
			NOT = {
				owner = {
					OR = {
						tag = SEJ
						was_tag = SEJ
					}
				}
			}
		}
		remove_province_modifier = THE_SEJ_p1_mayan_spread_center_modifier
	}
	if = {
		limit = { has_global_flag = THE_SEJ_downfall_revolution_active_global_flag }
		every_country = {
			limit = { THE_SEJ_downfall_revolution_active_st = yes }
			country_event = { id = THE_SEJ_events.155 }
		}
	}
}

on_battle_won_province = {
	if = {
		limit = {
			from = {
				tag = REB
				is_rebel_type = religious_rebels
			}
			owner = {
				OR = {
					tag = SEJ
					was_tag = SEJ
				}
				has_disaster = THE_SEJ_p1_fifty_nine_theses_disaster
				has_country_flag = THE_SEJ_p1_guard_missionary_mechanic_enabled_flag
			}
			NOT = { has_province_modifier = THE_SEJ_p1_guard_missionary_strength_modifier }
		}
		add_province_modifier = {
			name = "THE_SEJ_p1_guard_missionary_strength_modifier"
			duration = 1095
		}
	}
}

on_admiral_recruited = {
	if = {
		limit = {
			has_country_flag = THE_SEJ_p1_ruozhejiandui_admiral_tradition_flag
		}
		add_navy_tradition = 1
	}
}

on_colony_established = {
	if = {
		limit = { has_province_flag = THE_SEJ_ai_boss_frontier_province_flag }
		clr_province_flag = THE_SEJ_ai_boss_frontier_province_flag
	}
	if = {
		limit = {
			continent = south_america
			has_terrain = jungle
			owner = {
				OR = {
					AND = {
						OR = {
							tag = SEJ
							was_tag = SEJ
						}
						has_country_flag = THE_SEJ_p1_four_color_dyes_south_america_dyes_flag
					}
					colonial_parent = {
						OR = {
							tag = SEJ
							was_tag = SEJ
						}
						has_country_flag = THE_SEJ_p1_four_color_dyes_south_america_dyes_flag
					}
				}
			}
		}
		random = {
			chance = 5
			change_trade_goods = dyes
		}
	}
}
