THE_AKI_monoriko_monthly_effects = {
	#丰饶年度增长
	if = {
		limit = { 
			OR = {
				ROOT = {
					has_government_attribute = THE_AKI_harvest_autumn_province_buff_abilities
				}
				ROOT = {
					religion = the_minoriko
				}
			}
		}
		every_province = {
			limit = {
				owned_by = ROOT
				religion = the_minoriko
			}
			#导出省份发展度
				#此变量
			export_to_variable = {
				variable_name = the_aki_religions_province_development_var
				value = development
			}
				#x-2
			subtract_variable = {
				which = the_aki_religions_province_development_var
				value = 2
			}
				#98/x
			set_variable = {
				which = the_aki_religions_province_development_var_2
				value = 98
			}
			divide_variable = {
				which = the_aki_religions_province_development_var_2
				which = the_aki_religions_province_development_var
			}
			#改变丰饶增长
			change_variable = {
				which = the_aki_religions_monoriko_province_im_var
				which = the_aki_religions_province_development_var_2
			}
		}
	}
	#年度丰饶检测100 
	if = {
		limit = { 
			OR = {
				ROOT = {
					has_government_attribute = THE_AKI_harvest_autumn_province_buff_abilities
				}
				ROOT = {
					religion = the_minoriko
				}
			}
		}
		every_province = {
			limit = {
				owned_by = ROOT
				religion = the_minoriko
				check_variable = {
					which = the_aki_religions_monoriko_province_im_var
					value = 100
				}
			}
			add_base_tax = 1
			add_base_production = 1
			add_base_manpower = 1
			change_variable = {
				which = the_aki_religions_monoriko_province_im_var
				value = -100
			}
		}
	}
	
}
