	if = {
		limit = { has_idea = THE_NIN_colorfulegg_eu4_idea_7 }
		every_country = {
			add_aggressive_expansion = {
				who = ROOT
				value = -1000
				apply_calc = no
			}
		}
	}
}

THE_NIN_on_marine_raised_scripted_effect = {
	if = {
		limit = { has_idea = THE_NIN_colorfulegg_eu4_idea_5 }
		hidden_effect = {
			change_variable = {
				which = the_nin_eu4_marine_var
				value = 1
			}
			if = {
				limit = { check_variable = { which = the_nin_eu4_marine_var value = 50 } }
				subtract_variable = {
					which = the_nin_eu4_marine_var
					value = 50
				}
				if = {
					limit = {
						adm = 6
						dip = 6
						mil = 6
					}
					add_prestige_or_monarch_power = {
						amount = 200
					}
				}
				else = {
					change_adm = 1
					change_dip = 1
					change_mil = 1
				}
			}
		}
	}
}
