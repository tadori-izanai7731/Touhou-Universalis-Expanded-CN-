THE_NIN_on_get_new_province_effect = {
	if = {
		limit = { 
			owner = {
				tag = NIN
				has_country_modifier = THE_NIN_the_victory_godness_modifier
			}
		}
		change_culture = greek
	}
}

THE_NIN_change_cloth_to_silk_province_effect = {
	if = {
		limit = { 
			trade_goods = cloth
			owner = {
				mission_completed = THE_NIN_4_5
			}
		}
		change_trade_goods = silk
		owner = {
			add_mercantilism = 2
		}
	}
	if = {
		limit = {
			trade_goods = fish
			owner = {
				mission_completed = THE_NIN_2_5
			}
		}
		change_trade_goods = THE_pearls_tradegoods
		add_base_production = 2
	}
}

THE_NIN_won_war_with_ven_effect = {
	if = {
		limit = {
			tag = NIN
			has_country_flag = THE_NIN_cannot_call_ally_VEN_flag
			FROM = {
				tag = VEN
			}
		}
		clr_country_flag = THE_NIN_cannot_call_ally_VEN_flag
	}
}

THE_NIN_on_start_change_price_effect = {
	if = {
		limit = {
			tag = NIN
			has_country_modifier = THE_NIN_black_curtain_modifier
		}
		change_price = {
			trade_goods = THE_pearls_tradegoods
			key = THE_pearls_tradegoods_shortage_1
			value = 1.0
			duration = 7300
		}
	}
}

THE_NIN_add_institution_progress_effect = {
	if = {
		limit = {
			owner = {
				mission_completed = THE_NIN_3_5
			}
		}
		THE_NIN_add_institution_progress_script = { institution = renaissance }
		THE_NIN_add_institution_progress_script = { institution = new_world_i }
		THE_NIN_add_institution_progress_script = { institution = printing_press }
		THE_NIN_add_institution_progress_script = { institution = global_trade }
		THE_NIN_add_institution_progress_script = { institution = manufactories }
		THE_NIN_add_institution_progress_script = { institution = enlightenment }
		THE_NIN_add_institution_progress_script = { institution = industrialization }
	}
}
THE_NIN_add_institution_progress_script = {
	if = {
		limit = {
			is_institution_enabled = $institution$
		}
		add_institution_embracement = {
			which = $institution$
			value = 10
		}
	}
}

THE_NIN_develop_province_to_add_prestige_effect = {
	if = {
		limit = { 
			owner = {
				tag = NIN
				has_government_attribute = THE_NIN_art_and_venus_reform_mechanic
			}
		}
		owner = {
			add_government_power = {
				mechanic_type = THE_NIN_art_and_venus_reform_mechanic_ex
				power_type = THE_NIN_art_and_venus_reform_mechanic_ex_power
				value = 1
			}
		}
	}
	if = {
		limit = { 
			owner = {
				overlord = {
					tag = NIN
					has_government_attribute = THE_NIN_art_and_venus_reform_mechanic
				}
			}
		}
		owner = {
			overlord = {
				add_government_power = {
					mechanic_type = THE_NIN_art_and_venus_reform_mechanic_ex
					power_type = THE_NIN_art_and_venus_reform_mechanic_ex_power
					value = 1
				}
			}
		}
	}
}

THE_NIN_check_prestige_monthly_event_effect = {
	if = {
		limit = { 
			has_government_attribute = THE_NIN_fictional_foam_reform_mechanic
		}
		if = {
			limit = { 
				prestige = 50
				NOT = {
					has_government_power = {
						mechanic_type = THE_NIN_fictional_foam_reform_mechanic_ex
						power_type = THE_NIN_fictional_foam_reform_mechanic_ex_power
						value = 4
					}
				}
			}
			custom_tooltip = THE_NIN_check_prestige_monthly_event_effect_c_tt1_yes
			custom_tooltip = THE_NIN_check_prestige_monthly_event_effect_c_tt3
		}
		else = {
			custom_tooltip = THE_NIN_check_prestige_monthly_event_effect_c_tt1_no
			custom_tooltip = THE_NIN_check_prestige_monthly_event_effect_c_tt3
		}
	}
	else = {
		custom_tooltip = THE_NIN_check_prestige_monthly_event_effect_c_tt2
	}
}

THE_NIN_add_government_power_foam_effect = {
	add_government_power = {
		mechanic_type = THE_NIN_fictional_foam_reform_mechanic_ex
		power_type = THE_NIN_fictional_foam_reform_mechanic_ex_power
		value = 1
	}
}
