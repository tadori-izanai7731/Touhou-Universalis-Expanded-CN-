					add_government_reform = steppe_horde
					change_religion = animism
					THE_SEJ_ai_boss_apply_far_effect = yes
				}
			}
		}
		if = {
			limit = { exists = CSU }
			CSU = {
				if = {
					limit = { ai = yes }
					set_primitive = no
					THE_SEJ_ai_boss_apply_far_effect = yes
				}
			}
		}
		if = {
			limit = { exists = INC }

				if = {
					limit = {
						ai = yes
						was_tag = CSU
					}

			}
